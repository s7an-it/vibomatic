/** Supported social media platforms for signal scraping. */
export type Platform = 'twitter' | 'reddit' | 'hackernews' | 'linkedin' | 'youtube';

/** AI/ML subfield categories for trend classification. Minimum 8 per TRN-04. */
export type TrendCategory =
  | 'nlp'
  | 'computer-vision'
  | 'mlops'
  | 'reinforcement-learning'
  | 'generative-ai'
  | 'data-engineering'
  | 'deep-learning'
  | 'statistics-math'
  | 'robotics'
  | 'ai-safety';

/** Lifecycle status of a trend based on signal trajectory. */
export type TrendStatus = 'hot' | 'rising' | 'stable' | 'fading';

/** Platform-level signal contribution to a trend. */
export interface PlatformSignal {
  platform: Platform;
  mentionCount: number;
  latestSignalAt: Date;
}

/**
 * A synthesized trend representing a currently-discussed AI/ML topic.
 * Produced by TrendAnalyzer from aggregated ScrapedSignals.
 * Serves TRN-01, TRN-02, TRN-03, TRN-04.
 */
export interface Trend {
  id: string;
  topic: string;
  summary: string;
  whyTrending: string | null;
  subfield: TrendCategory;
  signalStrength: number; // 0-100 composite score
  status: TrendStatus;
  platforms: PlatformSignal[];
  firstSeen: Date;
  lastUpdated: Date;
  scrapeCycleId: string | null;
}

/** User's binary relevance rating for a trend. Serves TRN-06. */
export type TrendRating = 'relevant' | 'not_relevant';

export interface TrendFeedback {
  id: string;
  userId: string;
  trendId: string;
  rating: TrendRating;
  createdAt: Date;
}

/** Refresh status returned by GET /api/trends/refresh-status. Serves TRN-05. */
export interface TrendRefreshStatus {
  lastRefreshedAt: Date | null;
  nextScheduledAt: Date | null;
  platformStatuses: Record<Platform, 'healthy' | 'degraded' | 'disabled'>;
}

import type { TrendCategory } from './trend';

/** Format of a learning resource. Serves LRN-03. */
export type ResourceFormat = 'tutorial' | 'paper' | 'video' | 'repo' | 'course' | 'article';

/** Difficulty level for resources and skill assessments. */
export type Difficulty = 'beginner' | 'intermediate' | 'advanced';

/** User's binary quality rating for a resource. Serves TRN-07. */
export type ResourceFeedback = 'helpful' | 'not_helpful';

/** Progress status for a user working through a resource. Serves SKL-02. */
export type ProgressStatus = 'not_started' | 'in_progress' | 'completed';

/**
 * A learning resource discovered by ContentAggregator.
 * Displayed in learning paths. Serves LRN-01, LRN-03, LRN-05.
 */
export interface LearningResource {
  id: string;
  trendId: string | null;
  title: string;
  url: string;
  format: ResourceFormat;
  difficulty: Difficulty;
  estimatedMinutes: number | null;
  source: string;
  qualityScore: number; // 0.0 - 1.0, adjusted by feedback
  prerequisites: SkillPrerequisite[];
  createdAt: Date;
}

/** A prerequisite skill requirement for a resource. Serves LRN-05. */
export interface SkillPrerequisite {
  category: SkillCategory;
  minimumLevel: Difficulty;
}

/**
 * A personalized, ordered collection of resources for a specific trend.
 * Generated by RecommendationEngine. Serves LRN-01, LRN-02, LRN-04.
 */
export interface LearningPath {
  trendId: string;
  userId: string;
  resources: LearningPathEntry[];
  generatedAt: Date;
  expiresAt: Date;
}

/** A single entry in a learning path with user-specific context. */
export interface LearningPathEntry {
  resource: LearningResource;
  position: number;
  progress: ProgressStatus;
  completedAt: Date | null;
  feedback: ResourceFeedback | null;
  prerequisiteMet: boolean; // false triggers "prerequisite needed" indicator per LRN-05
}

/**
 * User progress record for a specific resource.
 * Written on completion. Serves SKL-02, SKL-04.
 */
export interface UserProgress {
  id: string;
  userId: string;
  resourceId: string;
  trendId: string | null;
  status: ProgressStatus;
  startedAt: Date | null;
  completedAt: Date | null;
  feedback: ResourceFeedback | null;
}

/** Cached learning path to avoid repeated LLM calls. */
export interface CachedLearningPath {
  id: string;
  trendId: string;
  userSkillHash: string;
  resourceIds: string[];
  generatedAt: Date;
  expiresAt: Date;
}

// packages/shared/src/types/platform.ts

import type { Platform } from './trend';

/**
 * A raw signal collected from a social media platform.
 * Stored in scraped_signals table. Input to TrendAnalyzer.
 * Serves SIG-01, SIG-02, SIG-04.
 */
export interface ScrapedSignal {
  id: string;
  platform: Platform;
  originalUrl: string | null;
  title: string | null;
  bodyText: string | null; // truncated to 2000 chars
  authorId: string | null;
  engagementCount: number;
  commentCount: number;
  contentHash: string; // SHA-256 for dedup per SIG-02
  topicExtracted: string | null;
  isRelevant: boolean | null;
  relevanceMethod: 'keyword' | 'llm' | null;
  publishedAt: Date | null;
  scrapedAt: Date;
  scrapeCycleId: string | null;
}

/**
 * Normalized signal format that all platform adapters must produce.
 * Common interface per SIG-01.
 */
export interface NormalizedSignal {
  platform: Platform;
  originalUrl: string;
  title: string;
  bodyText: string;
  authorId: string;
  engagementCount: number;
  commentCount: number;
  publishedAt: Date;
}

/**
 * Interface that all platform adapters must implement.
 * One adapter per platform (5 total). Serves SCR-01.
 */
export interface PlatformAdapterInterface {
  readonly platform: Platform;
  readonly isEnabled: boolean;

  /** Scrape signals from the last N hours. Returns normalized signals. */
  scrape(lookbackHours: number): Promise<NormalizedSignal[]>;

  /** Validate that API credentials are working. */
  validateCredentials(): Promise<boolean>;

  /** Report current rate limit status. */
  getRateLimitStatus(): RateLimitStatus;
}

/** Rate limit tracking per adapter. Serves SCR-05. */
export interface RateLimitStatus {
  platform: Platform;
  requestsRemaining: number;
  resetAt: Date;
  isThrottled: boolean;
}

/** Configuration for the scrape orchestrator. Serves SCR-03, SCR-06. */
export interface ScrapeConfig {
  enabledPlatforms: Platform[];
  cadenceHours: number; // default 24
  lookbackHours: number; // default 48
  retentionDays: number; // default 30, serves SIG-03
  maxConcurrentAdapters: number;
}

/** Result of a completed scrape cycle. */
export interface ScrapeCycleResult {
  cycleId: string;
  startedAt: Date;
  completedAt: Date;
  platformResults: PlatformScrapeResult[];
  totalSignals: number;
  newSignals: number; // after dedup
  relevantSignals: number; // after filtering
}

/** Per-platform result within a scrape cycle. Serves SCR-04. */
export interface PlatformScrapeResult {
  platform: Platform;
  status: 'success' | 'partial' | 'failed';
  signalCount: number;
  errorMessage: string | null;
  durationMs: number;
}

/**
 * Standard API response envelope.
 * All endpoints return this shape per tech design response conventions.
 */
export interface ApiResponse<T> {
  data: T;
  meta: ApiMeta | null;
  errors: ApiError[] | null;
}

/** Pagination and metadata for list endpoints. */
export interface ApiMeta {
  total: number;
  limit: number;
  offset: number;
  timestamp: string; // ISO 8601
}

/**
 * Error response following RFC 7807 Problem Details.
 * Returned for all error conditions.
 */
export interface ApiError {
  type: string; // URI reference identifying the problem type
  title: string;
  status: number; // HTTP status code
  detail: string;
  instance?: string; // URI of the specific occurrence
}

/** Query parameters for GET /api/trends. Serves TRN-01, TRN-04. */
export interface TrendListParams {
  subfield?: string;
  status?: string;
  limit?: number; // default 10
  offset?: number; // default 0
}

/** Request body for POST /api/trends/:id/rate. Serves TRN-06. */
export interface TrendRateBody {
  rating: 'relevant' | 'not_relevant';
}

/** Request body for POST /api/progress. Serves SKL-02. */
export interface ProgressUpdateBody {
  resourceId: string;
  status: 'not_started' | 'in_progress' | 'completed';
}

/** Request body for POST /api/progress/:resourceId/feedback. Serves TRN-07. */
